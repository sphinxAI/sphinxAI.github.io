[문서]    def getContainerMask(self,image,dx,dy,cx,cy,angle) :
        ny,nx,nc = image.shape
        maskContainer = np.zeros((ny,nx,3))
        maskContainer[int(ny/2-dy/2):int(ny/2+dy/2),int(nx/2-dx/2):int(nx/2+dx/2),:] = 1
        if angle != 0.0 :
            maskContainer = self.getRotatedImage(maskContainer,angle)
        return self.getTranslatedImage(maskContainer,cx-nx/2,cy-ny/2).astype(np.uint8)

[문서]    def getAngle(self,image,x1,y1,x2,y2,dh,ifDebug) :
        ny,nx,nc = image.shape
        getY = lambda x : next((y for y in range(ny) if image[y,x,0] < self.thSetAngle), ny-1)
        nDiv = 10
        
        xt = [float(nDiv-iDiv)/(nDiv+1)*x1+float(iDiv+1)/(nDiv+1)*x2 for iDiv in range(nDiv)]
        yt = [getY(int(xt[iDiv])) for iDiv in range(nDiv)]
        A = np.vstack([xt, np.ones(len(xt))]).T

        dydx,y0 = np.linalg.lstsq(A, yt,rcond=-1)[0]     
        y1, y2 = dydx*x1+y0, dydx*x2+y0
        dx, dy = x2 - x1, y2 - y1
        dl = pow(dx*dx+dy*dy,0.5)
        angle = 180.0/3.1415926536*np.arctan(-dy/dx)
        dhx, dhy = dh*dx/dl,dh*dy/dl
        x3, y3 = x2 - dhy, y2 + dhx
        x4, y4 = x1 - dhy, y1 + dhx
        cx, cy = (x1+x2+x3+x4)/4, (y1+y2+y3+y4)/4
        if ifDebug :
            print(' dl=',dl,'dydx= ',dydx,' point 1 = ',x1,y1,'point 2 = ',x2,y2, 'point 3 = ',x3,y3,'point 4 = ',x4,y4)
        
        return x3,y3,x4,y4,cx,cy,angle,dl

[문서]    def getContainerImage(self,image,xEnd1,dl,dh,cx,cy,angle,ifDebug) :
        imageBk,maskBlack = self.extBackgroundMask(500,image,xEnd1,ifDebug)
        image = np.clip(imageBk - image,0,self.maxImgV).astype(np.uint8)
        _,mask = cv2.threshold(image,self.thContainerMask,1,cv2.THRESH_BINARY)
        
        maskContainer = self.getContainerMask(image,dl,dh,cx,cy,angle)
    
        image = ((self.maxImgV - (image*mask))*maskBlack)*maskContainer
        
        image = self.extContainerImage(image,dl,dh,cx,cy,angle)     
        return image
